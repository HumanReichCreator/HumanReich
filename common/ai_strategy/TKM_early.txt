TKM_North_has_nothing_to_do_with_us = {
	enable = {
		tag = TKM
    has_war_with = WAK
		OR = {
			OR = {
			 	is_in_faction_with = CIR
	     	CIR = { is_subject_of = TKM }
			}
			OR = {
			 	is_in_faction_with = SEJ
	     	SEJ = { is_subject_of = TKM }
			}
			OR = {
			 	is_in_faction_with = NIB
	     	NIB = { is_subject_of = TKM }
			}
			OR = {
			 	is_in_faction_with = RMI
	     	RMI = { is_subject_of = TKM }
			}
			OR = {
			 	is_in_faction_with = MSD
	     	MSD = { is_subject_of = TKM }
			 }
		}

	}
	abort = {
		OR = {
		  WAK = { has_capitulated = yes }
			NOT = { country_exists = WAK }
		  not = { WAK = { is_subject_of = SCA } }
		}
	}
	
	abort_when_not_enabled = yes
	ai_strategy = {
		type = dont_defend_ally_borders
		id = "CIR"
		value = 100
	}
	ai_strategy = {
		type = dont_defend_ally_borders
		id = "SEJ"
		value = 100
	}
	ai_strategy = {
		type = dont_defend_ally_borders
		id = "NIB"
		value = 100
	}
	ai_strategy = {
		type = dont_defend_ally_borders
		id = "RMI"
		value = 100
	}
	ai_strategy = {
		type = dont_defend_ally_borders
		id = "MSD"
		value = 100
	}

}

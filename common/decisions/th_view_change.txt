                limit = {
                    check_variable = { OL_view_type = 0}

                limit = {
                    check_variable = { OL_view_type = 2}

            factor = 0

                check_variable = { OL_view_type = 1 }

                check_variable = { OL_view_value > 85}
                factor = 0
            }
            modifier = {
                go_weixin_condition = yes
                add = 5

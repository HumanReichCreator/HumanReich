			tag = ROOT
			has_war_with = ROOT
			has_opinion = {
              target = PREV
              value < -100
            }
		}
    	or = {
	    	ROOT = {
		    	has_country_flag = use_by_focus_@PREV
	    	}
	    	and = {
				or = {
			        and = {
			        	or = {
			        		is_in_faction_with = ROOT
			        		is_subject_of = ROOT
			        		ROOT = { is_subject_of = PREV}
			        	}
			        }
			        and = {
			        	has_country_flag = {
			                flag = openvalue
			                value > -1
			            }
			            ROOT = {
			                has_country_flag = {
				                flag = openvalue     
				                value > -1
			                }
			            } 
			            has_opinion = { target = ROOT value > 39 }  
			        }
			        has_opinion = { target = ROOT value > 99 }  
				    
				}
				ROOT = {
					not = { has_country_flag = owns_core_state@PREV }
				}
		    }     
    	}	
	}

country_event = {
	#xuanwuchuan win root
	id = spoil_event.1
	title = spoil_event.1.t
	desc = spoil_event.1.d
	fire_only_once = no
	is_triggered_only = no
	trigger = {
		ROOT = {
			has_country_flag = Spoil_xiaxuanwucheng
		}
		or = {
			and = {
				69 = {
					is_controlled_by = ROOT
				}
				69 = {
					is_owned_by = var:ROOT.spoil_xiaxuanwucheng_target
				}
			}
			var:ROOT.spoil_xiaxuanwucheng_target = {
				surrender_progress > 0.7
			}
		}
	}
	mean_time_to_happen = {
		days = 2
	}
	option = {
		#军工
		name = spoil_event.1.1
		clr_country_flag = Spoil_xiaxuanwucheng
		#remove_mission = Spoil_mission_xiaxuanwucheng
		hidden_effect = {
			remove_mission = micFact5_maintain
			remove_mission = micFact2_maintain
			remove_mission = micFact1_maintain
		}
		if
		{
			limit = {
				has_idea = SGS_mic_spoil0_idea
			}
			swap_ideas = {
				remove_idea = SGS_mic_spoil0_idea
				add_idea = SGS_mic_spoil2_idea
			}
		}
		else_if = {
			limit = {
				has_idea = SGS_mic_spoil1_idea
			}
			swap_ideas = {
				remove_idea = SGS_mic_spoil1_idea
				add_idea = SGS_mic_spoil5_idea
			}
			#activate_mission = micFact5_maintain
		}
		else_if = {
			limit = {
				has_idea = SGS_mic_spoil2_idea
			}
			swap_ideas = {
				remove_idea = SGS_mic_spoil2_idea
				add_idea = SGS_mic_spoil5_idea
			}
		}
		else_if = {
			limit = {
				has_idea = SGS_mic_spoil5_idea
			}
			remove_mission = micFact5_maintain
		}
		else = {
			add_ideas = SGS_mic_spoil2_idea
		}
		var:ROOT.spoil_xiaxuanwucheng_target = {
			#white_peace = ROOT
			country_event = {
				id = spoil_event.101
				days = 1
				random = 1
			}
		}
		ai_chance = {
			base = 35
		}
	}
	option = {
		#装备
		name = spoil_event.1.2
		clr_country_flag = Spoil_xiaxuanwucheng
		#remove_mission = Spoil_mission_xiaxuanwucheng
		var:ROOT.spoil_xiaxuanwucheng_target = {
			hidden_effect = {
				add_equipment_to_stockpile = {
					type = buqiang_a
					amount = 500
				}
			}
			send_equipment = {
				equipment = buqiang_a
				amount = 1000
				target = ROOT
			}
			country_event = {
				id = spoil_event.101
				days = 1
				random = 1
			}
		}
		ai_chance = {
			base = 25
		}
	}
}

country_event = {
	#xuanwuchuan win root
	id = spoil_event.34
	title = spoil_event.34.t
	desc = spoil_event.1.d
	fire_only_once = no
	is_triggered_only = no
	trigger = {
		ROOT = {
			has_country_flag = Spoil_chongcun
		}
		or = {
			and = {
				106 = {
					is_controlled_by = ROOT
				}
				106 = {
					is_owned_by = var:ROOT.spoil_chongcun_target
				}
			}
			var:ROOT.spoil_chongcun_target = {
				surrender_progress > 0.7
			}
		}
	}
	mean_time_to_happen = {
		days = 2
	}
	option = {
		#军工
		name = spoil_event.1.1
		clr_country_flag = Spoil_chongcun
		hidden_effect = {
			remove_mission = micFact5_maintain
			remove_mission = micFact2_maintain
			remove_mission = micFact1_maintain
		}
		if
		{
			limit = {
				has_idea = SGS_mic_spoil0_idea
			}
			swap_ideas = {
				remove_idea = SGS_mic_spoil0_idea
				add_idea = SGS_mic_spoil1_idea
			}
		}
		else_if = {
			limit = {
				has_idea = SGS_mic_spoil1_idea
			}
			swap_ideas = {
				remove_idea = SGS_mic_spoil1_idea
				add_idea = SGS_mic_spoil2_idea
			}
			#activate_mission = micFact5_maintain
		}
		else_if = {
			limit = {
				has_idea = SGS_mic_spoil2_idea
			}
			swap_ideas = {
				remove_idea = SGS_mic_spoil2_idea
				add_idea = SGS_mic_spoil5_idea
			}
		}
		else_if = {
			limit = {
				has_idea = SGS_mic_spoil5_idea
			}
			remove_mission = micFact5_maintain
		}
		else = {
			add_ideas = SGS_mic_spoil1_idea
		}
		var:ROOT.spoil_chongcun_target = {
			#white_peace = ROOT
			country_event = {
				id = spoil_event.134
				days = 1
				random = 1
			}
		}
		ai_chance = {
			base = 35
		}
	}
	option = {
		#装备
		name = spoil_event.1.2
		clr_country_flag = Spoil_chongcun
		var:ROOT.spoil_chongcun_target = {
			hidden_effect = {
				add_equipment_to_stockpile = {
					type = buqiang_a
					amount = 125
				}
				add_equipment_to_stockpile = {
					type = spell_card_a
					amount = 125
				}
			}
			send_equipment = {
				equipment = buqiang_a
				amount = 250
				target = ROOT
			}
			send_equipment = {
				equipment = spell_card_a
				amount = 250
				target = ROOT
			}
			country_event = {
				id = spoil_event.134
				days = 1
				random = 1
			}
		}
		ai_chance = {
			base = 25
		}
	}
}

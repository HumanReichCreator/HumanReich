country_event = {
	#拉鹰
	id = sys.1
	title = sys.1.t
	desc = sys.1.desc

	fire_only_once = no
	is_triggered_only = yes

	option = {
		if = {
			limit = {
				fandong < 0.5
			}
			add_popularity = {
				ideology = fandong
				popularity = 0.03
			}
			else = {
				add_popularity = {
					ideology = fandong
					popularity = 0.01
				}
			}
		}
		
		if = {
			limit = {
				not = { has_government = fandong}
			}
			add_stability = -0.04
		}

		name = sys.1.a
		if = {
			limit = {
				not = { has_government = oldorder }
			}
			go_fandong = yes
		}

		if = {
			limit = {
				has_idea = fandong_drift1
			}
			modify_timed_idea = {
				idea = fandong_drift1 days = 92
			}
			else = {
				add_timed_idea = { idea = fandong_drift1 days = 92}
			}
		}
		

		ai_chance = {
			base = 1
			modifier = {
				has_government = fandong
				factor = 2.5
			}
			modifier = {
				has_country_flag = go_fandong
				factor = 1000
			}
			modifier = {
				date > 150.1.1
				has_government = fandong
				factor = 100
			}
		}
	}

	option = {
		if = {
			limit = {
				neworder < 0.5
			}
			add_popularity = {
				ideology = neworder
				popularity = 0.03
			}
			else = {
				add_popularity = {
					ideology = neworder
					popularity = 0.01
				}
			}
		}

		if = {
			limit = {
				not = { has_government = neworder}
			}
			add_stability = -0.04
		}

		name = sys.1.b

		if = {
			limit = {
				not = { has_government = oldorder }
			}
			go_neworder = yes
		}

		if = {
			limit = {
				has_idea = neworder_drift1
			}
			modify_timed_idea = {
				idea = neworder_drift1 days = 92
			}
			else = {
				add_timed_idea = { idea = neworder_drift1 days = 92}
			}
		}
		

		ai_chance = {
			base = 1
			modifier = {
				has_government = neworder
				factor = 2.5
			}
			modifier = {
				has_country_flag = go_neworder
				factor = 1000
			}
			modifier = {
				date > 150.1.1

country_event = {
	#判断有没有核心
	id = sys.7
	hidden = yes
	fire_only_once = no
	is_triggered_only = no
	option = {}
	mean_time_to_happen = {
		days = 2
	}
	trigger = {
		any_other_country = {
			ROOT = {
				any_owned_state = {
					is_core_of = PREV.PREV
				}
			}
		}
		
	}
	immediate = {
		every_country = {
			limit = {
				ROOT = {
					any_owned_state = {
						is_core_of = PREV.PREV
					}
				}
				not = { tag = ROOT }
			}
			ROOT = {
				set_country_flag = owns_core_state@PREV
			}
		}
	}
}

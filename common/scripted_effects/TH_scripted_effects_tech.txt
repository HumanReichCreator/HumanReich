tech_initialize = {
   if = {
     limit = {
	   OR = {
	     has_tech = waike
	   }
	 }
     waike = yes
   }
   if = {
     limit = {
	   OR = {
	     has_tech = lingben
	   }
	 }
     lingben = yes
   }
   if = {
     limit = {
	   NOT = {
	     has_tech = waike
		 has_tech = lingben
	   }
	 }
     add_ideas = none_theory
   }
   ##############
   if = {
     limit = {
        has_idea = Beast
     }
     set_technology = { beast_tech = 1 }	 
   }
   if = {
     limit = {
        has_idea = Fairy
